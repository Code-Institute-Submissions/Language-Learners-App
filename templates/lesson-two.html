Lesson 2: Loops and Contionals

If and Else Statements

if(condition) {
// code gets executed, if true
}
else if(condition) {
// code gets executed, if first condition is not true
}
else {
// code gets executed, if all else is false
}

Example:

let minutes = 30;
let hour = 1;
if (minutes == 2 && hour != 1) {
return "Two minutes have passed."
}
else if(minutes == 26 || hours == 1) {
return "More than 25 minutes have passed."
}
else {
return "Less than two minutes have passed"
}
The result will be "More than 25 minutes have passed.", as hour = 1.

Switch Statements

switch(expression) {
case x:
// code gets executed in case x matches
break; // required to stop the execution
case y:
// code is here
continue; // Means that the case will be skipped, if matched
case z:
// code gets executed, if true
break;
default:
// code that gets executed in case nothing of the above matches
}

For loops

for (start; finish; increment/decrement) {
// Code comes here
}

let arr = [3,4,5,6,7]

for(i = 0; i < arr.length; i++) {
// Code goes here
}
